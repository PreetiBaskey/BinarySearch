//approach - 1
//time complexity - O(logn)
//space complexity - O(1)
class Solution {
public:
    int getPivotIndex(vector<int> &nums) {
        int n = nums.size();

        int left = 0, right = n - 1;
        while(left < right) {

            while(left < right && nums[left] == nums[left + 1]) {
                left++;
            }
            while(left < right && nums[right] == nums[right - 1]) {
                right--;
            }

            int mid = left + (right - left) / 2;

            if(nums[mid] > nums[right]) {
                left = mid + 1;
            }
            else {
                right = mid;
            }
        }

        return right;
    }
    bool gotTargetIndex(vector<int> &nums, int left, int right, int target) {
        int n = nums.size();
        bool result = false;

        while(left <= right) {
            int mid = left + (right - left) / 2;

            if(nums[mid] < target) {
                left = mid + 1;
            }
            else if(nums[mid] > target) {
                right = mid - 1;
            }
            else {
                result = true;
                break;
            }
        }

        return result;
    } 
    bool search(vector<int>& nums, int target) {
        int n = nums.size();

        int pivotIndex = getPivotIndex(nums);

        bool result = gotTargetIndex(nums, 0, pivotIndex - 1, target);

        if(!result) {
            result = gotTargetIndex(nums, pivotIndex, n - 1, target);
        }

        return result;
    }
};
