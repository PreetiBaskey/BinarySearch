//approach - 1
//time complexity - O(logn)
//space complexity - O(1)
class Solution {
public:
    bool isPossible(vector<int> &time, int totalTrips, long long midTime) {
        long long currentTotalTime = 0;
        for(int &t: time) {
            currentTotalTime += midTime / t;
        }

        return currentTotalTime >= totalTrips;
    }
    long long minimumTime(vector<int>& time, int totalTrips) {
        long long left = 1;
        int minTime = *min_element(begin(time), end(time));
        long long right = (long long)minTime * totalTrips;

        while(left < right) {
            long long midTime = left + (right - left) / 2;
            if(isPossible(time, totalTrips, midTime)) {
                right = midTime;
            }
            else {
                left = midTime + 1;
            }
        }

        return left;
    }
};
