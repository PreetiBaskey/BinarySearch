#include<bits/stdc++.h>
using namespace std;

int getPivot(vector<int> &nums) {
    int n = nums.size();
    
    int left = 0, right = n - 1;
    while(left < right) {
        int mid = left + (right - left) / 2;
        
        if(nums[mid] > nums[right]) {
            left = mid + 1;
        }
        else {
            right = mid;
        }
    }
    
    return right;
}

int main() {
    vector<int> nums = {4, 5, 1, 2, 3};
    int n = nums.size();    
    
    int pivotIndex = getPivot(nums);
    
    int result = n - pivotIndex;
    
    cout<<"number of times the array rotated is :- "<<result<<endl;
    
    return 0;
}
