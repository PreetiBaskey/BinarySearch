//approach - 1
//time complexity - O(logn)
//space complexity - O(1)
class Solution {
public:
    int getPivotIndex(vector<int> &nums) {
        int n = nums.size();
        
        int left = 0, right = n - 1;
        while(left < right) {
            int mid = left + (right - left) / 2;
            if(nums[mid] > nums[right]) {
                left = mid + 1;
            }
            else {
                right = mid;
            }
        }

        return right;
    }
    int findTargetInLeft(vector<int> &nums, int left, int right, int target) {
        int n = nums.size();
        int idx = -1;

        while(left <= right) {
            int mid = left + (right - left) / 2;

            if(nums[mid] < target) {
                left = mid + 1;
            }
            else if(nums[mid] > target) {
                right = mid - 1;
            }
            else {
                idx = mid;
                break;
            }
        }

        return idx;
    }
    int findTargetInRight(vector<int> &nums, int left, int right, int target) {
        int idx = -1;

        while(left <= right) {
            int mid = left + (right - left) / 2;

            if(nums[mid] < target) {
                left =  mid + 1;
            }
            else if(nums[mid] > target) {
                right = mid - 1;
            }
            else {
                idx = mid;
                break;
            }
        }

        return idx;
    }
    int search(vector<int>& nums, int target) {
        int n = nums.size();

        int pivotIndex = getPivotIndex(nums);

        int idx = findTargetInLeft(nums, 0, pivotIndex - 1, target);
        if(idx == -1) {
            idx = findTargetInRight(nums, pivotIndex, n - 1, target);
        }

        return idx;
    }
};
